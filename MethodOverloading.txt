// Demonstration of Method Overloading (Compile-time Polymorphism)

class Calculator {
    private int result;

    // Overloaded method for integers
    int add(int a, int b) {
        result = a + b;
        return result;
    }

    // Overloaded method for doubles
    double add(double a, double b) {
        return a + b;
    }

    // Getter method for integer result
    public int getResult() {
        return result;
    }
}

public class Main {
    public static void main(String[] args) {
        Calculator c = new Calculator();

        System.out.println("Int result: " + c.add(5, 10));
        System.out.println("Double result: " + c.add(2.5, 3.5));
    }
}
