import java.util.*;

public class Main {
    public static void main(String args[]) {
        // Create a Scanner object to read user input
        Scanner sc = new Scanner(System.in);
        
        // Prompt the user for the number of elements
        System.out.print("Enter Number of elements: ");
        
        // Read the number of elements
        int n = sc.nextInt();
        
        // Declare and instantiate the array
        int[] arr = new int[n];
        
        // Prompt the user to enter array elements
        System.out.print("Enter array elements: ");
        
        // Read the array elements from the user
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        
        // --- Insertion Sort Algorithm ---
        // Start from the second element (index 1) and consider it the key
        for (int i = 1; i < n; i++) {
            // Store the current element to be inserted into the sorted subarray
            int key = arr[i];
            
            // Start comparing with the element just before the key
            int j = i - 1;
            
            // Move elements of arr[0..i-1] that are greater than key
            // to one position ahead of their current position
            while (j >= 0 && arr[j] > key) { // Corrected condition: arr[j] > key
                arr[j + 1] = arr[j];
                j--;
            }
            
            // Place the key element in its correct sorted position
            arr[j + 1] = key;
        }
        
        // --- Print the sorted array ---
        System.out.print("Sorted: ");
        for (int x : arr) {
            System.out.print(x + " ");
        }
        
        // Close the scanner object
        sc.close();
    }
}
