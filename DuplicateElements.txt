public class Main {
    public static void main(String[] args) {
        int[] array = {6, 2, 8, 1, 4, 2, 7, 9, 5, 1, 7, 4};
        System.out.println("Duplicate elements found:");

        for (int i = 0; i < array.length - 1; i++) {
            for (int j = i + 1; j < array.length; j++) {
                if (array[i] == array[j]) {
                    System.out.println(array[i]);
                }
            }
        }
    }
}
